\begin{frame}{Instrumentierung - Fork}
    \begin{itemize}
      \item \texttt{signal(ts, nr)} in Trace
      \item Erzeuge neuen Trace
      \item Starte der neuen Routine
      \item \texttt{wait(ts, nr)} in neuen Trace
      \item Ausführen der Routine
    \end{itemize}
\end{frame}

\begin{frame}{Instrumentierung - Mutex}
  \begin{itemize}
    \item (Try)(R)Lock:
    \begin{itemize}
      \item \texttt{lock(ts, mu.id, ⟨-|t|r|tr⟩, ⟨0|1⟩)}
      \item Führe Lock aus
    \end{itemize}
    \item (Try)(R)Unlock:
    \begin{itemize}
      \item \texttt{unlock(ts, mu.id)} in Trace
      \item Führe Lock aus
    \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}{Instrumentierung - Channel}
  \begin{itemize}
    \item Send: (c <- i):
    \begin{itemize}
      \item \texttt{pre(ts, c.id, true)} in Trace
      \item \texttt{c <- \{i, r.id, ts\}}
      \item \texttt{post(ts', c.id, true, ts)} in Trace
    \end{itemize}
    \item Receive (a := <- c):
    \begin{itemize}
      \item \texttt{pre(ts, c.id, false)} in Trace
      \item \texttt{\{i, id\_send, ts\_send\} := <-c}
      \item \texttt{post(ts', ch.id, false, id\_send, ts\_send)} in Trace
      \item \texttt{a := i}
    \end{itemize}
    \item Close (close(c))
    \begin{itemize}
      \item \texttt{close(ts, c.id)} in Trace
      \item \texttt{close(c)}
    \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}{Analyse - Channel}
  \begin{itemize}
    \item $x$ ist Ursache von $y$ wenn
    \begin{itemize}
      \item $x \to y \Leftrightarrow VC(x) < VC(y) \Leftrightarrow \forall z (VC(x)_z \leq VC(y)_z) \land \exists z' (VC(x)_{z'} < VC(y)_{z'})$  
    \end{itemize}
    \item $x$ ist nebenläufig/unvergleichbar mit $y$ wenn
    \begin{itemize}
      \item $ x || y \Leftrightarrow VC(x) \not \lessgtr VC(y) \Leftrightarrow VC(x) \nless VC(y) \land VC(y) \nless VC(x)$
    \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}{Analyse - Channel}

    \end{itemize}
    \item Potenzielle Kommunikationspartner auf gebufferten Channels:
    \begin{itemize}

      \item $\#s_{i, <} \leq \#r_{j, <} + \#r_{i, \not\lessgtr}$
      \item $\#s_{i, <} + \#s_{i, \not\lessgtr} \geq \#r_{j, <}$
    \end{itemize}
  \end{itemize}
\end{frame}

      \item Pre- oder Post-VC von Send unvergleichbar mit VC von Close
      \item Wenn Send ist Ursache von Close $\Rightarrow$ Send auf Close tritt 
      während Durchlauf auf $\Rightarrow$ erzeugt Laufzeitfehler
    \end{itemize}
  \end{itemize}
\end{frame}
